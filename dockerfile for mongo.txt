################################################
#### installing mongodb using docker file
### base image from Ubuntu
################################################

FROM ubuntu:16.04

MAINTAINER Morderngeek

# Getting the public GPG Key 
RUN apt-key adv --keyserver hkp://keyserver.ubuntu.com:80 --recv 2930ADAE8CAF5059EE73BB4B58712A2291FA4AD5

#creating a list file
RUN echo "deb [ arch=amd64,arm64 ] https://repo.mongodb.org/apt/ubuntu xenial/mongodb-org/3.6 multiverse" | sudo tee /etc/apt/sources.list.d/mongodb-org-3.6.list

# Updating and installing
RUN apt-get update && apt-get install -y mongodb-org

#data directory
RUN mkdir -p /data/db

#preferred port is default
EXPOSE 27017

#for executing entry command
CMD ["--port 27017"]

#setting default command
ENTRYPOINT usr/bin/mongodb
